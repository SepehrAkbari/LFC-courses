Table patients {
  ssn varchar [pk]
  name varchar
  address varchar
  age int
  primaryPhysician_ssn varchar [ref: > doctors.ssn]
}

Table doctors {
  ssn varchar [pk]
  name varchar
  specialty varchar
  yearsOfExperience int
}

Table pharmaceuticalCompany {
  name varchar [pk]
  phoneNumber varchar
}

Table drug {
  tradeName varchar [pk]
  manufacturer varchar [ref: > pharmaceuticalCompany.name]
  formula varchar
  price decimal
  primary key (tradeName, manufacturer)
}

Table pharmacy {
  name varchar
  address varchar
  phoneNumber varchar
}

Table prescription {
  prescriptionId int [pk]
  date date
  quantity int
  patient_ssn varchar [ref: > patients.ssn]
  doctor_ssn varchar [ref: > doctors.ssn]
  drugTradeName varchar [ref: > drug.tradeName]
  drugManufacturer varchar [ref: > drug.manufacturer]
}

Table contract {
  contractId int [pk]
  startDate date
  contractText text
  pharmaceuticalCompanyName varchar [ref: > pharmaceuticalCompany.name]
  pharmacyName varchar [ref: > pharmacy.name]
}

Table supervisor {
  supervisorId int [pk]
  name varchar
  contractId int [ref: > contract.contractId]
}
